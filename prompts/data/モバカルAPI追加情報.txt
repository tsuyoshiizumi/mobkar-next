<?php
// ランダムは何でもよい
$random = "AAAAAAAAAAAAAAABBBBBBBBBBBBBBBBB");
// ランダムを秘密鍵(sha256)で暗号化
$prev_enc =hash_hmac('sha256', $random, "1234567890abcdefghijklmnopqrstuvwxyz",false);
// 暗号化したものをbase64でエンコード
$encoded = base64_encode($prev_enc);
echo "$encoded \n";

// ランダムもbase64でエンコード
$enc_random = base64_encode($random);
echo "$random \n";
echo "$enc_random \n";

?>
